% ── Compatibility patch: abntex2 1.9.7 + memoir 3.8+ ──
% \imprimirfolhaderosto passes unbraced tokens to \@ifstar, causing the next
% \input in main.tex to be consumed as an argument.  Redefine it with braced
% code blocks and correct optional-arg forwarding.
\makeatletter
\renewcommand{\imprimirfolhaderosto}[1][\folhaderostoname]{%
  \@ifstar
    {\begin{folhaderosto*}[#1]\folhaderostocontent\end{folhaderosto*}}%
    {\begin{folhaderosto}[#1]\folhaderostocontent\end{folhaderosto}}%
}
\makeatother
